    },
    loadOnlyText() {
      this.textGroup.position.y = 0
      this.scene.add(this.textGroup);
      this.textShowFlag = true
      this.is3DfileLoaded = true
    },
    animate() {
      var A = this;
      if (this.renderFlag) {
        requestAnimationFrame(() => {
          return A.animate()
        })
        this.controls.update()
        this.renderer.render(this.scene, this.camera)
        this.particalTime += .05;
        this.textGroup.children.forEach(((o, e) => {
          if (A.textShowFlag && (o.getObjectByName("circle").material.opacity < 1 && (o.getObjectByName("circle").material.opacity += .001),
            o.getObjectByName("text").material.opacity < 1 && (o.getObjectByName("text").material.opacity += .015),
            o.getObjectByName("circle").material.opacity >= 1 && o.getObjectByName("text").material.opacity >= 1 && (A.textShowFlag = !1,
              A.shinnied = !0)),
            A.shinnied) {
            A.particalTime += .001;
            var t = 1 - (Math.sin(A.particalTime) + 1) / 2 * (e % 8) / 16;
            o.getObjectByName("circle").material.opacity = t
          }
        }
        ))
      }

    },
    onSwipeChange(index) {
      this.currentIndicator = index;
    },
    convertToImage (container, options = {}) {
      // 设置放大倍数
      const scale = window.devicePixelRatio;

      // 传入节点原始宽高
      const _width = container.offsetWidth;
      const _height = container.offsetHeight;

      let { width, height } = options;
      width = width || _width;
      height = height || _height;

      // html2canvas配置项
      const ops = {
        scale,
        width,
        height,
        useCORS: true,
        allowTaint: false,
        ...options
      };

      function dataURLtoFile(dataURI, type) {
        let binary = atob(dataURI.split(',')[1]);
        let array = [];
        for(let i = 0; i < binary.length; i++) {
          array.push(binary.charCodeAt(i));
        }
        return new Blob([new Uint8Array(array)], {type:type });
      }
      
      return html2canvas(container, ops).then(canvas => {
        // 返回图片的二进制数据
        return dataURLtoFile(canvas.toDataURL("image/png"), "image/png");
      });
    },
    @debounce(1000)
    async getShareImg() {
      const file = await this.convertToImage(this.$refs.sharePage);
      const reader = new FileReader()
      reader.readAsDataURL(file)
      reader.onloadend = e => {
        const src = e.target.result
        const image = new Image()
        image.onload = () => {
          UploadToOss(file).then(data => {
            let url = data.res.requestUrls[0]
            if (url.indexOf('?') !== -1) {
              url = url.substring(0, url.indexOf('?'))
            }
            this.sharePNG = url + '?x-oss-process=image/resize,w_375,limit_0';
            this.handleShare();
          })
        }
        image.src = src
      };
    },
    @debounce(500)
    handleShare() {
      let shareObj = {};
      let img = {
        title: '推荐好书赢好礼！世界读书日让我们一起与书为伴~',
        image: this.sharePNG
      };
      (shareObj.data = [
        {
          type: 3,
          platformType: '2',
          ...img,
        },
      ]),
      this.$$callApp(
        'ls_open_share',
        {
          type: '999',
          ...shareObj,
        },
        function(res) {
          callback(JSON.parse(res));
        }
      );
    }
  },
};
